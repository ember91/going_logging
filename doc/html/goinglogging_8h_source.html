255 void print_prefix(const std::string& file, unsigned int line,
256  const std::string& func) noexcept {
257 #ifndef GOINGLOGGING_DISABLED
258  uint32_t cnt = 0;
259 
260  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
261  const char sep =
262 #ifdef _WIN32
263  '\\';
264 #else
265  '/';
266 #endif // _WIN32
267  size_t idx = file.find_last_of(sep);
268  std::cout << (idx == std::string::npos ? file : file.substr(idx + 1));
269  ++cnt;
270  }
271  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
272  if (cnt == 0) {
273  std::cout << "Line: ";
274  } else {
275  std::cout << ":";
276  }
277  std::cout << line;
278  ++cnt;
279  }
280  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
281  if (cnt != 0) {
282  std::cout << ", ";
283  }
284  std::cout << func << "()";
285  ++cnt;
286  }
287  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
288  if (cnt != 0) {
289  std::cout << ", ";
290  }
291  auto t = std::chrono::system_clock::now();
292  auto in_time_t = std::chrono::system_clock::to_time_t(t);
293  std::cout << std::put_time(std::localtime(&in_time_t), "%H:%M:%S");
294  ++cnt;
295  }
296  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
297  if (cnt != 0) {
298  std::cout << ", ";
299  }
300  std::cout << "TID: " << std::this_thread::get_id();
301  ++cnt;
302  }
303 
304  if (cnt != 0) {
305  std::cout << ": ";
306  }
307 #else
308  // Avoid warning of unused variable
309  (void)(file);
310  (void)(line);
311  (void)(func);
312 #endif // GOINGLOGGING_DISABLED
313 }
