330 std::ostream& operator<<(std::ostream& os, const Prefixer& p) noexcept {
332  uint32_t cnt = 0;
333 
334  // FILE
335  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
336  // Path separator
337  const char sep =
338 #ifdef _WIN32
339  '\\';
340 #else
341  '/';
342 #endif // _WIN32
343 
345  const char* path = p.get_file_path();
346 
347  // Get index of last file path separator
349  size_t str_len = std::strlen(path);
351  size_t sep_idx = std::numeric_limits<size_t>::max();
352  // No need to check for str_len = 0 here, since that will stop the for
353  // loop anyway
354  for (size_t i = str_len - 1; i != std::numeric_limits<size_t>::max();
355  --i) {
356  if (path[i] == sep) {
357  sep_idx = i;
358  break;
359  }
360  }
361 
362  // Output
363  os << (sep_idx == std::numeric_limits<size_t>::max() ?
364  path :
365  path + sep_idx + 1);
366 
367  ++cnt;
368  }
369 
370  // LINE
371  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
372  if (cnt == 0) {
373  os << "Line: ";
374  } else {
375  os << ":";
376  }
377  os << p.get_file_line_number();
378 
379  ++cnt;
380  }
381 
382  // FUNCTION
383  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
384  if (cnt != 0) {
385  os << ", ";
386  }
387  os << p.get_function_name() << "()";
388  ++cnt;
389  }
390 
391  // TIME
392  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
393  // Get time
394  auto now = std::chrono::system_clock::now();
395  std::time_t sinceEpoch = std::chrono::system_clock::to_time_t(now);
396  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
397  now.time_since_epoch()) %
398  1000;
399 
400  std::tm* local = std::localtime(&sinceEpoch);
401  if (local != nullptr) {
402  if (cnt != 0) {
403  os << ", ";
404  }
405  os << std::put_time(local, "%H:%M:%S") << '.' << std::setfill('0')
406  << std::setw(3) << ms.count();
407  ++cnt;
408  }
409  }
410 
411  // THREAD
412  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
413  if (cnt != 0) {
414  os << ", ";
415  }
416  os << "TID: " << std::this_thread::get_id();
417  ++cnt;
418  }
419 
420  // Final separator, if any
421  if (cnt != 0) {
422  os << ": ";
423  }
424 
425  return os;
426 }
