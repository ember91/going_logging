340 std::ostream& operator<<(std::ostream& os, const Prefixer& p) noexcept {
342  uint32_t cnt = 0;
343 
344  // FILE
345  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
346  // Path separator
347  const char sep =
348 #ifdef _WIN32
349  '\\';
350 #else
351  '/';
352 #endif // _WIN32
353 
355  const char* path = p.get_file_path();
356 
357  // Get index of last file path separator
358  size_t str_len = std::strlen(path);
359  size_t sep_idx = std::numeric_limits<size_t>::max();
360  // No need to check for str_len = 0 here, since that will stop the for loop anyway
361  for (size_t i = str_len - 1; i != std::numeric_limits<size_t>::max(); --i) {
362  if (path[i] == sep) {
363  sep_idx = i;
364  break;
365  }
366  }
367 
368  // Output
369  os << (sep_idx == std::numeric_limits<size_t>::max() ? path : path + sep_idx + 1);
370 
371  ++cnt;
372  }
373 
374  // LINE
375  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
376  if (cnt == 0) {
377  os << "Line: ";
378  } else {
379  os << ":";
380  }
381  os << p.get_file_line_number();
382 
383  ++cnt;
384  }
385 
386  // FUNCTION
387  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
388  if (cnt != 0) {
389  os << ", ";
390  }
391  os << p.get_function_name() << "()";
392  ++cnt;
393  }
394 
395  // TIME
396  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
397  // Get time
398  auto now = std::chrono::system_clock::now();
399  std::time_t sinceEpoch = std::chrono::system_clock::to_time_t(now);
400  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
401  now.time_since_epoch()) % 1000;
402 
403  std::tm* local = std::localtime(&sinceEpoch);
404  if (local != nullptr) {
405  if (cnt != 0) {
406  os << ", ";
407  }
408  os << std::put_time(local, "%H:%M:%S") << '.' <<
409  std::setfill('0') << std::setw(3) << ms.count();
410  ++cnt;
411  }
412  }
413 
414  // THREAD
415  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
416  if (cnt != 0) {
417  os << ", ";
418  }
419  os << "TID: " << std::this_thread::get_id();
420  ++cnt;
421  }
422 
423  // Final separator, if any
424  if (cnt != 0) {
425  os << ": ";
426  }
427 
428  return os;
429 }
