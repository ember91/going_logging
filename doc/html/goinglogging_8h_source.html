441 std::ostream& operator<<(std::ostream& os, const PrefixFormatter& p) noexcept {
443  uint32_t cnt = 0;
444 
445  // FILE
446  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
448  const char sep =
449 #ifdef _WIN32
450  '\\';
451 #else
452  '/';
453 #endif // _WIN32
454 
456  const char* path = p.get_file_path();
457 
458  // Get index of last file path separator
460  size_t str_len = std::strlen(path);
462  size_t sep_idx = std::numeric_limits<size_t>::max();
463  // No need to check for str_len = 0 here, since that
464  // will stop the for loop anyway
465  for (size_t i = str_len - 1; i != std::numeric_limits<size_t>::max();
466  --i) {
467  if (path[i] == sep) {
468  sep_idx = i;
469  break;
470  }
471  }
472 
473  // Output
474  os << (sep_idx == std::numeric_limits<size_t>::max() ?
475  path :
476  path + sep_idx + 1);
477 
478  ++cnt;
479  }
480 
481  // LINE
482  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
483  // Output 'Line' prefix only if file name has not been
484  // output
485  if (cnt == 0) {
486  os << "Line: ";
487  } else {
488  os << ":";
489  }
490  os << p.get_file_line_number();
491 
492  ++cnt;
493  }
494 
495  // FUNCTION
496  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
497  if (cnt != 0) {
498  os << ", ";
499  }
500  os << p.get_function_name() << "()";
501  ++cnt;
502  }
503 
504  // TIME
505  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
506  // Get time
507  auto now = std::chrono::system_clock::now();
508 
509  // Convert
510  std::time_t sinceEpoch = std::chrono::system_clock::to_time_t(now);
511  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
512  now.time_since_epoch()) %
513  1000;
514  std::tm* local = std::localtime(&sinceEpoch);
515 
516  if (local != nullptr) {
517  if (cnt != 0) {
518  os << ", ";
519  }
520  // Format and output
521  os << std::put_time(local, "%H:%M:%S") << '.' << std::setfill('0')
522  << std::setw(3) << ms.count();
523  ++cnt;
524  }
525  }
526 
527  // THREAD
528  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
529  if (cnt != 0) {
530  os << ", ";
531  }
532  os << "TID: " << std::this_thread::get_id();
533  ++cnt;
534  }
535 
536  // Final separator, if any
537  if (cnt != 0) {
538  os << ": ";
539  }
540 
541  return os;
542 }
