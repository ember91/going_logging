337 std::ostream& operator<<(std::ostream& os, const Prefixer& p) noexcept {
339  uint32_t cnt = 0;
340 
341  // FILE
342  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
343  // Path separator
344  const char sep =
345 #ifdef _WIN32
346  '\\';
347 #else
348  '/';
349 #endif // _WIN32
350 
351  // Convert to std::string
352  // TODO: This is unnecessary. Search for separator instead.
353  std::string file(p.get_file_path());
354 
355  // Start index of file name
356  size_t idx = file.find_last_of(sep);
357 
358  // Output
359  os << (idx == std::string::npos ? file : file.substr(idx + 1));
360 
361  ++cnt;
362  }
363 
364  // LINE
365  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
366  if (cnt == 0) {
367  os << "Line: ";
368  } else {
369  os << ":";
370  }
371  os << p.get_file_line_number();
372 
373  ++cnt;
374  }
375 
376  // FUNCTION
377  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
378  if (cnt != 0) {
379  os << ", ";
380  }
381  os << p.get_function_name() << "()";
382  ++cnt;
383  }
384 
385  // TIME
386  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
387  // Get time
388  auto now = std::chrono::system_clock::now();
389  std::time_t sinceEpoch = std::chrono::system_clock::to_time_t(now);
390  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
391  now.time_since_epoch()) % 1000;
392 
393  std::tm* local = std::localtime(&sinceEpoch);
394  if (local != nullptr) {
395  if (cnt != 0) {
396  os << ", ";
397  }
398  os << std::put_time(local, "%H:%M:%S") << '.' <<
399  std::setfill('0') << std::setw(3) << ms.count();
400  ++cnt;
401  }
402  }
403 
404  // THREAD
405  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
406  if (cnt != 0) {
407  os << ", ";
408  }
409  os << "TID: " << std::this_thread::get_id();
410  ++cnt;
411  }
412 
413  // Final separator, if any
414  if (cnt != 0) {
415  os << ": ";
416  }
417 
418  return os;
419 }
