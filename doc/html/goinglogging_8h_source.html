535 std::ostream& operator<<(std::ostream& os, const PrefixFormatter& p) noexcept {
537  uint32_t cnt = 0;
538 
539  // FILE
540  if ((curPrefixes & prefix::FILE) != prefix::NONE) {
542  const char sep =
543 #ifdef _WIN32
544  '\\';
545 #else
546  '/';
547 #endif // _WIN32
548 
550  const char* path = p.get_file_path();
551 
552  // Get index of last file path separator
554  size_t str_len = std::strlen(path);
556  size_t sep_idx = std::numeric_limits<size_t>::max();
557  // No need to check for str_len = 0 here, since that
558  // will stop the for loop anyway
559  for (size_t i = str_len - 1; i != std::numeric_limits<size_t>::max();
560  --i) {
561  if (path[i] == sep) {
562  sep_idx = i;
563  break;
564  }
565  }
566 
567  // Output
568  os << (sep_idx == std::numeric_limits<size_t>::max() ?
569  path :
570  path + sep_idx + 1);
571 
572  ++cnt;
573  }
574 
575  // LINE
576  if ((curPrefixes & prefix::LINE) != prefix::NONE) {
577  // Output 'Line' prefix only if file name has not been
578  // output
579  if (cnt == 0) {
580  os << "Line: ";
581  } else {
582  os << ":";
583  }
584  os << p.get_file_line_number();
585 
586  ++cnt;
587  }
588 
589  // FUNCTION
590  if ((curPrefixes & prefix::FUNCTION) != prefix::NONE) {
591  if (cnt != 0) {
592  os << ", ";
593  }
594  os << p.get_function_name() << "()";
595  ++cnt;
596  }
597 
598  // TIME
599  if ((curPrefixes & prefix::TIME) != prefix::NONE) {
600  // Get time
601  auto now = std::chrono::system_clock::now();
602 
603  // Convert
604  std::time_t sinceEpoch = std::chrono::system_clock::to_time_t(now);
605  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
606  now.time_since_epoch()) %
607  1000;
608  std::tm* local = std::localtime(&sinceEpoch);
609 
610  if (local != nullptr) {
611  if (cnt != 0) {
612  os << ", ";
613  }
614  // Format and output
615  os << std::put_time(local, "%H:%M:%S") << '.' << std::setfill('0')
616  << std::setw(3) << ms.count();
617  ++cnt;
618  }
619  }
620 
621  // THREAD
622  if ((curPrefixes & prefix::THREAD) != prefix::NONE) {
623  if (cnt != 0) {
624  os << ", ";
625  }
626  os << "TID: " << std::this_thread::get_id();
627  ++cnt;
628  }
629 
630  // Final separator, if any
631  if (cnt != 0) {
632  os << ": ";
633  }
634 
635  return os;
636 }
